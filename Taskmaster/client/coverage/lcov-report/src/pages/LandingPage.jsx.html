function AuthLink({ to, className, children, ...props }) {
  if (isAuthSite()) {
    return (
      <Link to={to} className={className} {...props}>
        {children}
      </Link>
    );
  }
  return (
    <a href={authUrl(to)} className={className} {...props}>
      {children}
    </a>
  );
}
