function getAllLighthouseRoutes() {
  const seen = new Set();
  return [...PUBLIC_ROUTES, ...PROTECTED_ROUTES].filter((r) => {
    if (seen.has(r.path)) return false;
    seen.add(r.path);
    return true;
  });
}
